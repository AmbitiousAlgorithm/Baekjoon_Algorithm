import itertools

def isPrime(n):
    prime = [True]*(n+1)
    prime[0], prime[1] = False, False
    
    for num in range(2, n+1):
        if not prime[num]:
            continue
        for mul in range(num*num, n+1, num):
            prime[mul] = False
    case = [idx for idx, val in enumerate(prime) if val]
    if n in case:
        return True
    return False
