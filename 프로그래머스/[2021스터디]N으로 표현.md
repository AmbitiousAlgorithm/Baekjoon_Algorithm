def solution(N, target):
    
    num_list = list({int(i*str(N))} for i in range(1,9))
    if N == target:
        return 1

    # Nì´ë‘ targetì´ë‘ ê°™ì€ ê²½ìš°?
    for i in range(1, len(num_list)):
        for j in range(i):
            for x in num_list[j]: 
                for y in num_list[i-j-1]:
                    num_list[i].add(x+y)
                    num_list[i].add(x-y)
                    num_list[i].add(x*y)
                    if y != 0:
                        num_list[i].add(x//y)
        if target in num_list[i]:
            return i+1
    
    return -1
