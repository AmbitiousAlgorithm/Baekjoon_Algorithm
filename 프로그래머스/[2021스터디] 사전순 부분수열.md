import itertools
import copy
def solution(s):
    answer = ''
    
    sentence = s 
    stack = [*s] 
    
    while len(stack) > 1:
        top = stack.pop()
        for s in copy.deepcopy(stack[::-1]): 
            part_list.append(top + s)
            
    answer_list = [*sentence] + part_list + [sentence]
    answer_list.sort()
    
    return answer
