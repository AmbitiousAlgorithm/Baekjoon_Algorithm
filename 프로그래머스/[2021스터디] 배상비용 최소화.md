```
import heapq

def solution(no, works):
    for i,v in enumerate(works):
        works[i] = -v
    heapq.heapify(works)
    
    while no >0:
        if works[0] == 0:
            break
        w = heapq.heappop(works)
        heapq.heappush(works,w+1)
        no -=1
        
    return sum(work*work for work in works)

```
