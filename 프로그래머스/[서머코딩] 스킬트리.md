```
import re

def solution(skill, skill_trees):
    answer = 0
    
    def check(word):
        compare =''
        for i in word:
            check = re.search(i,skill)
            if check != None:
                compare += check.group()
        pattern = re.compile(compare)
        if pattern.match(skill) != None:
            return 1
        else:
            return 0
    
    for i,word in enumerate(skill_trees):
        answer += check(skill_trees[i])
        
    return answer

```
